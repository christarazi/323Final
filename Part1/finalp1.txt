PROGRAM a2015;

(* This program computes and prints the value
			of an expression *)

var
(* declare variables *)

	a1 ,          ba ,            c, ba12   : integer   ;
BEGIN

		a1            = 3;
		ba =         4 ;
		 c           = 5   ;
		 print( 'a1=',   a1  ) ;
	print( ' value of c='    ,  c);
  (* compute the value of the expression *)
  		ba12 = a1 * ( ba + 2 * c)      ;
  		print( 'ba12 is equal to ',    ba12   );  (* print the value of ba12 *)

  	end. (* end of program *)
